# Error Contract

```rust
match agent.run("hello").await {
    Ok(out) => println!("{}", out.output_text),
    Err(AiError::RateLimit { retry_after }) => {
        tokio::time::sleep(retry_after).await;
    }
    Err(AiError::AuthFailed) => eprintln!("invalid api key"),
    Err(err) => eprintln!("unexpected: {err}"),
}
```
